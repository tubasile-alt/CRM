### 2) Consertar o clique (event delegation robusta)
Troque o handler por este (mais robusto):

document.addEventListener("click", async (e) => {
  const btn = e.target.closest("#btnScheduleTransplantSurgery");
  if (!btn) return;
  e.preventDefault();

  console.log("clicked schedule button");

  const date = prompt("Data da cirurgia (YYYY-MM-DD):");
  if (!date) return;

  const patientId = btn.getAttribute("data-patient-id") || window.__patientId;
  if (!patientId) {
    alert("Erro: patientId não encontrado.");
    return;
  }

  btn.disabled = true;
  try {
    const resp = await fetch(`/api/patient/${patientId}/transplant/schedule-surgery`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      credentials: "same-origin",
      body: JSON.stringify({ surgery_date: date })
    });

    const text = await resp.text();
    let out = null;
    try { out = JSON.parse(text); } catch {}

    if (!resp.ok) {
      console.error("HTTP error", resp.status, text);
      alert(`Falha ao agendar (HTTP ${resp.status}). Veja console.`);
      return;
    }

    if (!out || !out.success) {
      alert((out && out.error) ? out.error : "Erro ao agendar.");
      return;
    }

    if (out.calendar_event_created === false) {
      alert("Cirurgia agendada, mas falhou criar evento no Google Calendar. Veja logs.");
    } else {
      alert("Cirurgia agendada e evento criado no Google Calendar.");
    }

    location.reload();
  } catch (err) {
    console.error(err);
    alert("Erro inesperado ao agendar. Veja console.");
  } finally {
    btn.disabled = false;
  }
});
