CITE_RE = re.compile(r"\[cite:\s*\d+\]", re.IGNORECASE)

def strip_citations(value: str) -> str:
    if not value:
        return value
    cleaned = CITE_RE.sub("", value)
    cleaned = re.sub(r"\s{2,}", " ", cleaned)
    cleaned = re.sub(r"\s+([,.;:])", r"\1", cleaned)
    return cleaned.strip()
